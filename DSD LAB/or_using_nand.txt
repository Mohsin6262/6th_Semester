module NAND(O,A,B);
input A,B;
output O;
wire nand_out1;
wire nand_out2;
nand n1(nand_out1,A);
nand n2(nand_out2,B);
nand n3(O,nand_out1,nand_out2);
endmodule
module NAND_tb();
reg A,B;
wire O;
NAND in(O,A,B);
initial begin  //for multiple lines
A=0;
B=0;
#20 $display("A = %b",A, "B = %b",B,"O = %b",O);

A=0;
B=1;
#20 $display("A = %b",A, "B = %b",B,"O = %b",O);

A=1;
B=0;
#20 $display("A = %b",A, "B = %b",B,"O = %b",O);

A=1;
B=1;
#20 $display("A = %b",A, "B = %b",B,"O = %b",O);
end
endmodule
