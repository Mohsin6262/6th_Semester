//write a program which monitor a switch if it is not pressed then toggle LED attach with P1.0 if it is pressed then stop toggling.
#include <msp430.h> 

int main(void) {
    WDTCTL = WDTPW | WDTHOLD;   // Stop watchdog timer
    PM5CTL0 &= ~LOCKLPM5;       // Enable GPIOs

    P1DIR |= 0x01;              // Set P1.0 as output (LED)
    P1DIR &= ~0x04;             // Set P1.2 as input (Switch)
    P1OUT |= 0x04;              // Pull-up resistor on P1.2
    P1REN |= 0x04;              // Enable resistor on P1.2

    while(1) {
        if (P1IN & 0x04) {      // If switch is NOT pressed mean P1.2 will high
            P1OUT ^= 0x01;      // Toggle LED on P1.0
            __delay_cycles(100000); // Toggle speed
        }
    }
}
