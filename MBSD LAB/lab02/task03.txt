#include <msp430.h>

/*
 * main.c
 */
int main(void) {
    WDTCTL = WDTPW | WDTHOLD;	// Stop watchdog timer
    PM5CTL0 &= ~LOCKLPM5;       // Disable high-impedance mode to enable GPIOs
	P1DIR |= 0xFF;              //set P1 to output direction
	for(;;){
		volatile unsigned int i;
		P1OUT ^= 0xFF;    //toggle P1 using exclusive OR
		i=10000;
		do i--;       //do while loop for creating delay
		while(i!=0);
	}
}
